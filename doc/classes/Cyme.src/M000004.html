    def self.branches_from_cmdline(opts={})
        broker = opts[:broker] || 'amqp://'
        cmd = opts[:cmd] || 'cyme-list-branches'
        limit = opts[:limit]
        command = "#{cmd} --broker='#{broker}'"
        if limit:
            command = command << " --limit=#{limit}"
        end
        JSON.parse(%x[#{command}])
    end
