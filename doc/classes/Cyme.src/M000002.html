    def self.get_backend(backend=nil, prefix="Cyme::Backends", sep='::')
        backend ||= 'default'

        if backend.kind_of? String
            if backend.index(sep).nil?
                backend = [prefix, backend.capitalize()].join(sep)
            end
            return instantiate(backend)
        end

        backend
    end
