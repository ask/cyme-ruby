        def self.from_error(code, body)
            if code == 404
                return new code, 'NOT FOUND', body
            end
            if body.include? 'nok'
                error, traceback = JSON.parse(exc.http_body)['nok']
                new code, error, traceback
            else
                new code, nil, body
            end
        end
