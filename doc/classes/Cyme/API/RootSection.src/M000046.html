            def request(options={}, *path)
                method = options[:method]
                url = ([@url] + path).join('/').chomp('/') + '/'
                if @DEBUG
                    puts("-> #{method} #{url} #{options[:data]}")
                end
                @backend.request(method.to_s.downcase, url, options[:data])
            end
